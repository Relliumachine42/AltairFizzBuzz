                        function getValues() {
                            let fizzNumber = document.getElementById('fizzValue').value;
                            let buzzNumber = document.getElementById('buzzValue').value;
                            let endNumber = document.getElementById('endValue').value;
                            fizzNumber = Number(fizzNumber);
                            buzzNumber = Number(buzzNumber);
                            endNumber = Number(endNumber);
                            if(isNaN(fizzNumber) == true || isNaN(buzzNumber) == true || isNaN(endNumber) == true) {
                                Swal.fire({
                                    title: "I'm sorry!",
                                    text: 'We only accept numbers here.',
                                    icon: 'error',
                                    backdrop: false
                                });
                            } else if(endNumber > 10000) {
                                Swal.fire({
                                    title: '10,000 Limit',
                                    text: "You don't really need more than 10,000.",
                                    icon: 'error',
                                    backdrop: false
                                });
                            } else {
                                let contentArray = generateFizzBuzz(fizzNumber, buzzNumber, endNumber);
                                displayFizzBuzz(contentArray);
                            }
                        }
                        function generateFizzBuzz(fizz, buzz, end) {
                            let range = [];
                            for (let number = 1; number <= end; number++) {
                                if(number % fizz == 0 && number % buzz == 0) {
                                    range.push('FizzBuzz');
                                } else if(number % fizz == 0) {
                                    range.push('Fizz');
                                } else if(number % buzz == 0) {
                                    range.push('Buzz');
                                } else {
                                    range.push(number);
                                }
                            }
                            return range;
                        }
                        function displayFizzBuzz(contentToDisplay) {
                            let tableHtml = '';
                            for(let index = 0; index < contentToDisplay.length; index++) {
                                let currentContent = contentToDisplay[index];
                                let tableRowHtml = `<tr><td>${currentContent}</td></tr>`;
                                tableHtml += tableRowHtml;
                            }
                            document.getElementById('results').innerHTML = tableHtml;
                        }

The code is structured in three functions:

FizzBuzz

The entry point is the function, getValues(), which takes the values from the input forms on the Html. It then checks whether those values are numbers, displaying an error message if not. Otherwise, it checks whether the end number is greater than 10,000, displaying an error message in that case as well. Otherwise, it continues by intializing a variable contentArray with a function generateFizzBuzz() of three arguments, then runs a third function displayFizzBuzz() with the argument contentArray.

The second function, generateFizzBuzz(), is given three parameters, corresponding to the three arguments it was given inside the previous getValues() function. Then generateFizzBuzz() declares a variable range as an empty array. A for() loop examines each number in sequence from 1 to the value of end, pushing the number, fizz, buzz, or fizzbuzz into the array as determined by boolean logic and division remainders, and it increments the number by one and loops back, repeatedly until that number surpasses the value of the end, giving the array a processed version of every number from one to the end number.

The third function's, displayFizzBuzz(), parameter, contentToDisplay refers to the array generated by the second function. It declares the variable tableHtml, and using a second for() loop, it assigns the value of the array index to the variable currentContent starting at index 0 for the starting value. Variable tableRowHtml is initialized with a template literal consisting of html code that creates a table cell for each iteration of the variable currentContent in the for() loop, using string interpolation to insert the variable. tableHtml is added to tableRowHtml incrementing in the for() loop until it contains html for every table cell of every number between one and the end number. Finally, when the length of the array equals the index that's selecting the currentContent, it exits the for() loop and inserts the value of tableHtml into the table body element in the Html.
